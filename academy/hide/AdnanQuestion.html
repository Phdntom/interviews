# Here is a simple stacked procedure to carry out the above
# on any dataframe indexed out of dfclean by the 'Year' column.
def Poisson_Analysis(df):
    keep_cols = ['Journal', 'OnAc?']
    X = sm.add_constant(df[keep_cols])
    y = df['Citation Count']
    
    pois = sm.Poisson(y,X).fit(method='bfgs')
    
    print(pois.summary())
    print("\n")
    print("How unit change in parameters changes the citation rate?")
    print(np.exp(pois.params))
